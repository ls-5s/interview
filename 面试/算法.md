# dp

## 01 背包模板

每件物品，只能选一次，倒序遍历背包

```javascript

function knapsack01(weight, value, bagSize) {
    // dp[j] = 容量 j 的背包最大价值
    const dp = new Array(bagSize + 1).fill(0);

    // 遍历物品
    for (let i = 0; i < weight.length; i++) {
        // 重点：倒序遍历背包容量！！！
        for (let j = bagSize; j >= weight[i]; j--) {
            dp[j] = Math.max(dp[j], dp[j - weight[i]] + value[i]);
        }
    }

    return dp[bagSize];
}
```

## 完全背包模板

每件物品，能选无数次，正序遍历背包

```javascript
// 完全背包 标准模板
function completeKnapsack(weight, value, bagSize) {
    // dp[j] = 容量 j 的背包最大价值
    const dp = new Array(bagSize + 1).fill(0);

    // 遍历物品
    for (let i = 0; i < weight.length; i++) {
        // 重点：正序遍历背包容量！！！
        for (let j = weight[i]; j <= bagSize; j++) {
            dp[j] = Math.max(dp[j], dp[j - weight[i]] + value[i]);
        }
    }

    return dp[bagSize];
}
```

# 滑动窗口

我给你最简单、最通用、前端面试直接默写的滑动窗口模板！
一、万能模板（固定逻辑，所有题通用）

```js
var slidingWindow = function(s) {
    // 1. 定义窗口：左指针、哈希表/计数器
    let left = 0;
    let map = {}; // 记录窗口内元素出现次数

    // 2. 右指针遍历整个字符串/数组
    for(let right = 0; right < s.length; right++) {
        let c = s[right];
        // 把当前字符加入窗口
        map[c] = (map[c] || 0) + 1;

        // 3. 当窗口【不满足条件】时，移动左指针缩小窗口
        while(窗口不满足条件) {
            let d = s[left];
            map[d]--; // 移出窗口
            left++;   // 左指针右移
        }

        // 4. 窗口满足条件 → 更新答案
        结果 = Math.max(结果, right - left + 1);
    }
    return 结果;
}
```

# 双指针

```js
// 对撞指针模版
function twoPointers(nums, target) {
    let left = 0;
    let right = nums.length - 1;

    while (left < right) {
        const sum = nums[left] + nums[right];

        // 找到目标
        if (sum === target) {
            return [left, right]; // 返回结果
        }
        // 和太小，左指针右移
        else if (sum < target) {
            left++;
        }
        // 和太大，右指针左移
        else {
            right--;
        }
    }
    return [-1, -1]; // 未找到
}
```

# 数组去重

## 一维数组去重

1. 你问的：一维数组 去重
比如：[1,2,2,3,3,3]不用 JSON，直接用 Set 就行！

```js
// 一维数组去重（超级简单）
function unique(arr) {
  return [...new Set(arr)];
}

// 测试
console.log(unique([1,2,2,3,3])); // [1,2,3]
```

## 二维数组去重

可读性超高的 二维数组去重函数

```js
// 二维数组去重（清晰易懂版）
function unique(arr) {
    // 1. 把每个小数组变成字符串（方便去重）
    const strArr = arr.map(item => JSON.stringify(item));

    // 2. 用 Set 自动去掉重复的字符串
    const uniqueStrSet = new Set(strArr);
    
    // 3. 把 Set 转回数组
    const uniqueStrArr = Array.from(uniqueStrSet);
    
    // 4. 把字符串重新变回数组
    const result = uniqueStrArr.map(item => JSON.parse(item));
    
    return result;
}
```

# 前缀和

模版 1：基础前缀和数组（求子数组和）
适用场景
求区间和、子数组和判断、简单子数组问题

```js
// 1. 生成前缀和数组（固定写法）
const n = nums.length;
const prefix = new Array(n + 1).fill(0); // 关键：长度 n+1，prefix[0]=0

for (let i = 0; i < n; i++) {
    prefix[i+1] = prefix[i] + nums[i];
}

// 2. 求 [l, r] 子数组和（闭区间，从0开始）
const sum = prefix[r+1] - prefix[l];
```

模版 2：哈希表前缀和（最优解 O (n)）`
🔥 刷题最常用！解决：和为 K 的子数组个数

```js
var subarraySum = function(nums, k) {
    const map = new Map();
    map.set(0, 1); // 初始化：前缀和0 出现1次

    let sum = 0;
    let count = 0;
    
    for (const num of nums) {
        sum += num;
        
        // 核心：找之前出现过 sum - k
        if (map.has(sum - k)) {
            count += map.get(sum - k);
        }
        
        // 记录当前前缀和
        map.set(sum, (map.get(sum) || 0) + 1);
    }
    return count;
};
```

# 图论

DFS

```js
var numIslands = function(grid) {
    if (!grid || grid.length === 0) return 0;
    const rows = grid.length;
    const cols = grid[0].length;
    
    // 方向拆成两个一维数组
    const dx = [-1, 1, 0, 0];
    const dy = [0, 0, -1, 1];

    // 现在 dfs 传入 i 和 j 两个值
    const dfs = (i, j) => {
        // 越界 或 不是陆地，直接返回
        if (i < 0 || i >= rows || j < 0 || j >= cols || grid[i][j] === '0') {
            return;
        }

        // 标记为已访问
        grid[i][j] = '0';

        // 四个方向递归
        for (let k = 0; k < 4; k++) {
            dfs(i + dx[k], j + dy[k]);
        }
    };

    let count = 0;
    // 遍历每个格子
    for (let i = 0; i < rows; i++) {
        for (let j = 0; j < cols; j++) {
            if (grid[i][j] === '1') {
                count++;
                dfs(i, j); // 直接传 i,j 两个值
            }
        }
    }

    return count;
};
```

# 去重

```js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var topKFrequent = function(nums, k) {
    // 1. 统计每个数字出现的次数
    const freqMap = {};
    for (const num of nums) {
        freqMap[num] = (freqMap[num] || 0) + 1;
    }

    // 2. 数组去重（得到所有不重复的数字）
    const uniqueNums = [...new Set(nums)];

    // 3. 按出现次数从大到小排序
    uniqueNums.sort((a, b) => freqMap[b] - freqMap[a]);

    // 4. 取前 k 个
    return uniqueNums.slice(0, k);
};

```

# 回溯算法

```js
function 回溯函数() {
  if (满足结束条件) {
    保存结果
    return
  }

  for (遍历所有选择) {
    if (选择已经用过/不合法) continue

    做选择
    递归
    撤销选择 👈 这一步叫【回溯】
  }
}

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function(nums) {
    const res = []
    const path = []
    const used = []
    const bk = () => {
        if(path.length === nums.length) {
            res.push([...path])
            return
        }
        for(let i = 0;i < nums.length;i ++)
        {
            if(used[i]) continue
            used[i] = true
            path.push(nums[i])
            bk()
            used[i] = false
            path.pop();
        }
    }
    bk()
    return res

};
```

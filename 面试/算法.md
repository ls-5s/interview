# dp

## 01 背包模板

每件物品，只能选一次，倒序遍历背包

```javascript

function knapsack01(weight, value, bagSize) {
    // dp[j] = 容量 j 的背包最大价值
    const dp = new Array(bagSize + 1).fill(0);

    // 遍历物品
    for (let i = 0; i < weight.length; i++) {
        // 重点：倒序遍历背包容量！！！
        for (let j = bagSize; j >= weight[i]; j--) {
            dp[j] = Math.max(dp[j], dp[j - weight[i]] + value[i]);
        }
    }

    return dp[bagSize];
}
```

## 完全背包模板

每件物品，能选无数次，正序遍历背包

```javascript
// 完全背包 标准模板
function completeKnapsack(weight, value, bagSize) {
    // dp[j] = 容量 j 的背包最大价值
    const dp = new Array(bagSize + 1).fill(0);

    // 遍历物品
    for (let i = 0; i < weight.length; i++) {
        // 重点：正序遍历背包容量！！！
        for (let j = weight[i]; j <= bagSize; j++) {
            dp[j] = Math.max(dp[j], dp[j - weight[i]] + value[i]);
        }
    }

    return dp[bagSize];
}
```

# 滑动窗口

我给你最简单、最通用、前端面试直接默写的滑动窗口模板！
一、万能模板（固定逻辑，所有题通用）

```js
var slidingWindow = function(s) {
    // 1. 定义窗口：左指针、哈希表/计数器
    let left = 0;
    let map = {}; // 记录窗口内元素出现次数

    // 2. 右指针遍历整个字符串/数组
    for(let right = 0; right < s.length; right++) {
        let c = s[right];
        // 把当前字符加入窗口
        map[c] = (map[c] || 0) + 1;

        // 3. 当窗口【不满足条件】时，移动左指针缩小窗口
        while(窗口不满足条件) {
            let d = s[left];
            map[d]--; // 移出窗口
            left++;   // 左指针右移
        }

        // 4. 窗口满足条件 → 更新答案
        结果 = Math.max(结果, right - left + 1);
    }
    return 结果;
}
```

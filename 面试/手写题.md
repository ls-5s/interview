# 手写防抖 Debounce

```js
/**
 * 防抖函数 (Debounce)
 * 核心作用：高频触发的函数，只在【最后一次触发后等待指定时间】才执行一次
 * 应用场景：搜索框输入、窗口resize、按钮点击防重复提交
 * @param {Function} func - 需要防抖处理的目标函数
 * @param {Number} wait - 延迟执行的时间（毫秒）
 * @param {Boolean} immediate - 是否立即执行（true: 触发时立刻执行，等待期内不重复执行；false: 等待结束后执行）
 * @return {Function} 经过防抖处理后的函数
 */
function debounce(func, wait, immediate) {
  // 私有变量：闭包存储定时器ID，避免污染全局作用域
  var timeout;
  // 存储目标函数执行后的返回值（支持函数有返回值的场景）
  var result;

  // 闭包核心：返回防抖后的包装函数
  var debounced = function () {
    // 保存原函数的this指向（解决防抖后this指向丢失问题）
    var context = this;
    // 保存原函数的参数（解决防抖后参数丢失问题）
    var args = arguments;

    // 核心逻辑：如果定时器存在，清除上一次的定时器 → 重新计时
    if (timeout) clearTimeout(timeout);

    // ============== 立即执行逻辑 ==============
    if (immediate) {
      // callNow = 无定时器时为true → 代表可以立即执行
      // 第一次触发：timeout为undefined → !timeout = true → 执行
      // 等待期内触发：timeout存在 → !timeout = false → 不执行
      var callNow = !timeout;

      // 设置定时器：等待时间结束后，将timeout置为null → 允许下一次立即执行
      timeout = setTimeout(function () {
        timeout = null;
      }, wait);

      // 如果满足立即执行条件，调用原函数（绑定this和参数）
      if (callNow) result = func.apply(context, args);
    }
    // ============== 非立即执行逻辑（默认） ==============
    else {
      // 设置定时器：等待时间结束后，执行原函数
      timeout = setTimeout(function () {
        func.apply(context, args);
      }, wait);
    }

    // 返回原函数的执行结果
    return result;
  };
  // 返回包装后的防抖函数
  return debounced;
}
```

# 节流函数

```js
funtion throttle(func,wait) {
  var lastTime = 0
  var throttled = function() {
    var now = Date.now()
    var context = this
    var args = arguments
    if(now - lastTime >= wait) {
      func.apply(context,args)
      lastTime = now
    }
  }
  return throttled
}
```

Function.prototype.myapply = function (context,args) {
  context = context === null ? window : Object(context)
  const fn = Symbol()
  context[fn] = this
  let result
  try {
    result = args ? context[fn](...args) : context[fn]()
  } finally {
    delete context[fn]
  }
  return result
}

# 手写数组拍平 Array Flatten

```js
// 你写的 数组拍平函数（无限递归版）
function flatten(arr) {
  let res = []
  for (let item of arr) {
    // 如果当前元素是数组，递归拍平，再拼接到结果里
    if (Array.isArray(item)) {
      res = res.concat(flatten(item))
    } else {
      // 不是数组，直接放入结果
      res.push(item)
    }
  }
  return res
}

// ------------------- 测试 Demo -------------------
// 1. 定义一个【多层嵌套数组】（面试最常用的测试用例）
const nestedArr = [1, 2, [3, [4, 5, [6]]], 7, [[8]], 9]

// 2. 调用拍平函数
const flatArr = flatten(nestedArr)

// 3. 打印结果
console.log('原嵌套数组：', nestedArr)
console.log('拍平后数组：', flatArr)
```

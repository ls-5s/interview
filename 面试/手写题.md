# 手写防抖 Debounce

```js
/**
 * 防抖函数 (Debounce)
 * 核心作用：高频触发的函数，只在【最后一次触发后等待指定时间】才执行一次
 * 应用场景：搜索框输入、窗口resize、按钮点击防重复提交
 * @param {Function} func - 需要防抖处理的目标函数
 * @param {Number} wait - 延迟执行的时间（毫秒）
 * @param {Boolean} immediate - 是否立即执行（true: 触发时立刻执行，等待期内不重复执行；false: 等待结束后执行）
 * @return {Function} 经过防抖处理后的函数
 */
function debounce(func, wait, immediate) {
  // 私有变量：闭包存储定时器ID，避免污染全局作用域
  var timeout;
  // 存储目标函数执行后的返回值（支持函数有返回值的场景）
  var result;

  // 闭包核心：返回防抖后的包装函数
  var debounced = function () {
    // 保存原函数的this指向（解决防抖后this指向丢失问题）
    var context = this;
    // 保存原函数的参数（解决防抖后参数丢失问题）
    var args = arguments;

    // 核心逻辑：如果定时器存在，清除上一次的定时器 → 重新计时
    if (timeout) clearTimeout(timeout);

    // ============== 立即执行逻辑 ==============
    if (immediate) {
      // callNow = 无定时器时为true → 代表可以立即执行
      // 第一次触发：timeout为undefined → !timeout = true → 执行
      // 等待期内触发：timeout存在 → !timeout = false → 不执行
      var callNow = !timeout;

      // 设置定时器：等待时间结束后，将timeout置为null → 允许下一次立即执行
      timeout = setTimeout(function () {
        timeout = null;
      }, wait);

      // 如果满足立即执行条件，调用原函数（绑定this和参数）
      if (callNow) result = func.apply(context, args);
    }
    // ============== 非立即执行逻辑（默认） ==============
    else {
      // 设置定时器：等待时间结束后，执行原函数
      timeout = setTimeout(function () {
        func.apply(context, args);
      }, wait);
    }

    // 返回原函数的执行结果
    return result;
  };
  // 返回包装后的防抖函数
  return debounced;
}
```

```js
funtion throttle(func,wait) {
  var lastTime = 0
  var throttled = function() {
    var now = Date.now()
    var context = this
    var args = arguments
    if(now - lastTime >= wait) {
      func.apply(context,args)
      lastTime = now
    }
  }
  return throttled
}
```

# 手写数组拍平 Array Flatten

```js
// 你写的 数组拍平函数（无限递归版）
function flatten(arr) {
  let res = []
  for (let item of arr) {
    // 如果当前元素是数组，递归拍平，再拼接到结果里
    if (Array.isArray(item)) {
      res = res.concat(flatten(item))
    } else {
      // 不是数组，直接放入结果
      res.push(item)
    }
  }
  return res
}
```

# 手写 curry 函数，实现函数柯里化

```js
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args)
    }
    return function (...args2) {
      return curried.apply(this, args.concat(args2))
    }
  }
}
```

```js
function compose(...fns) {
```

# 使用 Vue3 Composable 组合式函数，实现 useRequest

```js
// useRequest.js (组合式函数文件)
import { ref } from 'vue'

// 封装异步请求 Hook
export function useRequest(apiFn) {
  // 1. 定义响应式状态
  const data = ref(null)   // 接口数据
  const loading = ref(false)// 加载状态
  const error = ref(null)  // 错误信息

  // 2. 核心请求方法
  const run = async () => {
    loading.value = true   // 开始请求
    error.value = null     // 清空错误
    try {
      const res = await apiFn() // 执行传入的接口函数
      data.value = res    // 赋值数据
    } catch (err) {
      error.value = err   // 捕获错误
    } finally {
      loading.value = false // 请求结束
    }
  }

  // 3. 返回状态和方法
  return {
    data,
    loading,
    error,
    run
  }
}
```

# 手写 LRU 缓存

```js
class LRUCache {
  constructor(capacity) {
    this.cache = new Map()
    this.capacity = capacity
  }

  get(key) {
    if (!this.cache.has(key)) return -1

    // 将访问的元素移到最新使用的位置
    const value = this.cache.get(key)
    this.cache.delete(key)
    this.cache.set(key, value)
    return value
  }

  put(key, value) {
    // 如果 key 已存在，先删除
    if (this.cache.has(key)) {
      this.cache.delete(key)
    }
    // 如果达到容量限制，删除最久未使用的元素
    else if (this.cache.size >= this.capacity) {
      // Map 的 keys() 会按插入顺序返回键
      const firstKey = this.cache.keys().next().value
      this.cache.delete(firstKey)
    }

    this.cache.set(key, value)
  }
}

// 使用示例
const cache = new LRUCache(2)
cache.put(1, 1) // 缓存是 {1=1}
cache.put(2, 2) // 缓存是 {1=1, 2=2}
console.log(cache.get(1)) // 返回 1
cache.put(3, 3) // 删除 key 2，缓存是 {1=1, 3=3}
console.log(cache.get(2)) // 返回 -1 (未找到)
```

# 手写 getType 函数

```js
// 定义一个通用的类型判断函数，接收任意数据 data
function getType(data) {
  // 核心：调用 Object 原型上的 toString 方法，精准判断数据类型
  // 返回结果格式：[object Null]、[object Array]、[object Object] 等
  const originType = Object.prototype.toString.call(data)

  // 截取字符串：从第8位开始，到倒数第1位结束
  // 把 [object Array] 截成 → Array
  const type = originType.slice(8, -1)

  // 转成小写，返回标准格式：array、null、object、function 等
  return type.toLowerCase()
}
```

# 深拷贝

```js
function deepClone(obj, map = new WeakMap()) {
  // 1. 非引用类型直接返回
  if (typeof obj !== 'object' || obj == null) return obj

  // 2. 解决循环引用：如果已经拷贝过，直接返回拷贝结果
  if (map.has(obj)) return map.get(obj)

  // 3. 区分数组/对象
  const res = Array.isArray(obj) ? [] : {}

  // 4. 缓存当前对象，防止循环引用
  map.set(obj, res)

  // 5. 递归拷贝
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      res[key] = deepClone(obj[key], map)
    }
  }

  return res
}
```

# 手写 new

```js
function mynew(Func, ...args) {
    // 1. 创建一个新对象（对应new第一步）
    const obj = {}
    // 2. 新对象原型指向构造函数原型对象（对应new第二步：继承原型）
    obj.__proto__ = Func.prototype
    // 3. 将构造函数的this指向新对象，执行构造函数（对应new第三步）
    let result = Func.apply(obj, args)
    // 4. 根据返回值判断（对应new第四步：处理返回值）
    return result instanceof Object ? result : obj
}
```
